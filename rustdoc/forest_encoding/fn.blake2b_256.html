pub fn blake2b_256(ingest: &[u8]) -> [u8; 32]
Expand description

Generates BLAKE2b hash of fixed 32 bytes size.

Example

use forest_encoding::blake2b_256;

let ingest: Vec<u8> = vec![];
let hash = blake2b_256(&ingest);
assert_eq!(hash.len(), 32);
